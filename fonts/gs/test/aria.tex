  \newcommand{\airout}[3]{%
    \pic (tap#2) at (#1,-2) {sean symbol=tap};
    \pic (nee#2) at (#1,-4) {sean symbol=needle};
    \draw[seg/analog] (tap#2-out) -- (nee#2-in);

  \draw[seg/analog] (s1-o4) -- (8.5, 0.15) -- (tapD-in);
