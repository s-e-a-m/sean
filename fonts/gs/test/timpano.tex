% ----------------------------------------------------------------
% \gschannel{prefix}{y}{label}
%   Posa la catena di canale: gmic→preamp→switch→hpf→invert→lsf→comp
%   con pic names #{prefix}mic, #{prefix}pre, #{prefix}sw,
%                  #{prefix}hpf, #{prefix}inv, #{prefix}lsf, #{prefix}cmp
% ----------------------------------------------------------------
\newcommand{\gschannel}[3]{%
  % trasduttore di ingresso
  \pic[rotate=90] (#1mic) at (0,{#2+1.2}) {sean symbol=gmic};
  \node[font=\footnotesize, anchor=south] at (0,{#2+1.85}) {#3};
  % catena
  \pic (#1pre) at (2,{#2})   {sean symbol=preamp};
  \pic (#1sw)  at (4,{#2})   {sean symbol=switch};
  \pic (#1hpf) at (6.5,{#2}) {sean symbol=hpf};
  \pic (#1inv) at (8.5,{#2}) {sean symbol=invert};
  \pic (#1lsf) at (10.5,{#2}){sean symbol=lsf};
  \pic (#1cmp) at (12.5,{#2}){sean symbol=comp};
  % connessioni interne al canale
  \draw[seg/analog] (#1mic-out)  -- (#1pre-in);
  \draw[seg/analog] (#1pre-out)  -- (#1sw-in);
  \draw[seg/analog] (#1sw-out)   -- (#1hpf-in);
  \draw[seg/analog] (#1hpf-out)  -- (#1inv-in);
  \draw[seg/analog] (#1inv-out)  -- (#1lsf-in);
  \draw[seg/analog] (#1lsf-out)  -- (#1cmp-in);
}
