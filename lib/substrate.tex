% --- vocabolario astratto (identità + ancore) ---
\def\sean@allsymbols{}
\newcommand{\seandeclaresymbol}[2]{%
  \ifcsname sean@sym@#1\endcsname\else
    \edef\sean@allsymbols{\ifx\sean@allsymbols\empty\else\sean@allsymbols,\fi#1}%
  \fi
  \expandafter\gdef\csname sean@anchors@#1\endcsname{#2}%
  \expandafter\gdef\csname sean@sym@#1\endcsname{1}%
}
\newcommand{\seanifsymbol}[3]{\ifcsname sean@sym@#1\endcsname#2\else#3\fi}
\newcommand{\seansymbolanchors}[1]{\csname sean@anchors@#1\endcsname}

% --- glifi ---
\newcommand{\seanglyph}[3]{\expandafter\gdef\csname sean@glyph@#1@#2\endcsname{#3}}
% --- font + parent (fallback) ---
\pgfkeys{/sean/font/parent/.initial={}}
\newcommand{\seanfont}[2]{%
  \pgfkeys{/sean/font/.cd,parent={},#2}%
  \expandafter\xdef\csname sean@parent@#1\endcsname{\pgfkeysvalueof{/sean/font/parent}}%
}
\newcommand{\seanparentof}[1]{\ifcsname sean@parent@#1\endcsname\csname sean@parent@#1\endcsname\fi}
% --- metadati di provenienza per-font ---
\pgfkeys{/sean/meta/.cd, author/.initial={}, year/.initial={}, note/.initial={}}
\newcommand{\seanfontmeta}[2]{%

  \expandafter\xdef\csname sean@meta@year@#1\endcsname{\pgfkeysvalueof{/sean/meta/year}}%
  \expandafter\xdef\csname sean@meta@note@#1\endcsname{\pgfkeysvalueof{/sean/meta/note}}}
\newcommand{\seanfontauthor}[1]{\ifcsname sean@meta@author@#1\endcsname\csname sean@meta@author@#1\endcsname\fi}
\newcommand{\seanfontyear}[1]{\ifcsname sean@meta@year@#1\endcsname\csname sean@meta@year@#1\endcsname\fi}

% --- font attivo ---
\def\sean@active{wb}
\newcommand{\seanusefont}[1]{\def\sean@active{#1}}
% --- risoluzione con fallback ---
\newcommand{\seanresolve}[2]{\def\sean@resolved{}\edef\sean@cur{#1}\sean@resolveloop{#2}}
\newcommand{\sean@resolveloop}[1]{%
  \ifx\sean@cur\empty\else
    \ifcsname sean@glyph@\sean@cur @#1\endcsname
      \let\sean@resolved\sean@cur
    \else
      \edef\sean@cur{\seanparentof{\sean@cur}}%
      \sean@resolveloop{#1}%
    \fi
  \fi}

% --- binding ancora dentro un glifo (pic): il nome pic viene prepeso da TikZ ---
% accumula anche il nome posato in \sean@posed (locale al gruppo: usato dal check)
\def\sean@posed{}
\newcommand{\seananchor}[2]{\coordinate (-#1) at #2;\edef\sean@posed{\sean@posed,#1}}
% --- override del font per-segno (vuoto = usa il font attivo); locale allo scope del pic ---
\def\sean@startfont{}
\tikzset{sean font/.code={\def\sean@startfont{#1}}}
% --- pic che risolve ed esegue il glifo (font attivo, o override 'sean font') ---
\tikzset{
  sean symbol/.pic={%
    \seanifsymbol{#1}{}{\PackageError{sean}{identità '#1' non dichiarata (\string\seandeclaresymbol)}{}}%
    \ifx\sean@startfont\empty\let\sean@from\sean@active\else\let\sean@from\sean@startfont\fi
    \seanresolve{\sean@from}{#1}%
    \ifx\sean@resolved\empty
      \seanmissing{#1}%
    \else
      \csname sean@glyph@\sean@resolved @#1\endcsname
    \fi}}
% segnaposto per glifo mancante: box tratteggiato etichettato + warning
\newcommand{\seanmissing}[1]{%
  \draw[dashed] (-0.5\seanunit,-0.5\seanunit) rectangle (0.5\seanunit,0.5\seanunit);
  \node[font=\tiny] at (0,0) {#1};
  \PackageWarning{sean}{nessun glifo per '#1' nel font attivo o nei suoi genitori}}

% --- verifica di conformità: tutte le ancore dichiarate sono posate dal glifo? ---
\newcommand{\seancheckglyph}[2]{%
  \begin{scope}[opacity=0]%
    \def\sean@posed{}%
    \csname sean@glyph@#1@#2\endcsname
    \edef\sean@list{\seansymbolanchors{#2}}%
    \foreach \a in \sean@list {%
      \in@{,\a,}{,\sean@posed,}%
      \ifin@\else\PackageWarning{sean}{glifo #1/#2: ancora '\a' dichiarata ma non posata}\fi}%
  \end{scope}%
  \message{^^JSEAN-OK: checked #1/#2^^J}}
