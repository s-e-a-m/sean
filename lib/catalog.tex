% Catalogo auto-generato dal registro (identità × font caricati).
% Ogni cella è una mini-tikzpicture indipendente dentro un nodo: così i glifi
% (anche quelli circuitikz) disegnano sempre vicino all'origine — niente
% "Dimension too large" quando la lista è lunga.
% Si usa il contatore nativo di \foreach (count=... from 0): un \pgfmathsetmacro
% dentro \foreach è locale al gruppo dell'iterazione e si azzererebbe.
\newcommand{\seancatalogcompare}[1]{%
  \begin{tikzpicture}[every node/.append style={font=\footnotesize}]
    \foreach \f [count=\c from 0] in {#1}{%
      \node at ({\c*2.6},1.4) {\textbf{\f}};}%
    \foreach \id [count=\r from 0] in \sean@allsymbols {%
      \node[anchor=east] at (-1.1,{-\r*2.2}) {\id};%
      \foreach \f [count=\c from 0] in {#1}{%
        \node at ({\c*2.6},{-\r*2.2}) {\tikz{\pic[sean font=\f] {sean symbol=\id};}};}}%
  \end{tikzpicture}}
% Vista per-font: un segno per riga + provenienza in testa.
\newcommand{\seancatalogfont}[1]{%
  \seanusefont{#1}%
  \begin{tikzpicture}[every node/.append style={font=\footnotesize}]
    \node[anchor=west] at (-1.1,2) {\textbf{#1} — \seanfontauthor{#1}, \seanfontyear{#1}};
    \foreach \id [count=\r from 0] in \sean@allsymbols {%
      \node at (0,{-\r*2.2}) {\tikz{\pic {sean symbol=\id};}};%
      \node[anchor=west] at (1.6,{-\r*2.2}) {\id};}%
  \end{tikzpicture}}
